CREATE TABLE axe_actualite (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    link VARCHAR(511) NOT NULL UNIQUE,
    pub_date DATE,
    content TEXT,
    source_name VARCHAR(255),
    is_selected TINYINT(1) DEFAULT 0
);
